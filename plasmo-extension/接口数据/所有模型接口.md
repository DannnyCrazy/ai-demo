### curl

curl 'https://casterfind.com/api/ProductBrowse/SearchByKeys' \
 -H 'Accept: application/json, text/plain, _/_' \
 -H 'Accept-Language: zh-CN,zh;q=0.9' \
 -H 'Connection: keep-alive' \
 -H 'Content-Type: application/json' \
 -H 'Origin: https://casterfind.com' \
 -H 'Referer: https://casterfind.com/productPc/' \
 -H 'Sec-Fetch-Dest: empty' \
 -H 'Sec-Fetch-Mode: cors' \
 -H 'Sec-Fetch-Site: same-origin' \
 -H 'User-Agent: Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/143.0.0.0 Safari/537.36' \
 -H 'sec-ch-ua: "Google Chrome";v="143", "Chromium";v="143", "Not A(Brand";v="24"' \
 -H 'sec-ch-ua-mobile: ?0' \
 -H 'sec-ch-ua-platform: "Windows"' \
 --data-raw '{"series":"10B","wheelMaterial":"PR-人造胶","surfaceColour":"褐","Abbreviation":"10B_PRPS","Lang":"Chinese"}'
